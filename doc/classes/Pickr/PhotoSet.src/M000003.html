                def self.get(id)
                        set  = flickr.photosets.getPhotos :photoset_id => id
                        info = flickr.photosets.getInfo   :photoset_id => id

                        PhotoSet.new(id, info.title,  info.description,
                                                           set.primary, set.photo)
                end
