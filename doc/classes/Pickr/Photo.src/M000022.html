                def to_html(img_size='thumb')
                        case img_size
                        when 'square'
                                img_url = to_square_url
                        when 'thumb'
                                img_url = to_thumbnail_url
                        when 'medium'
                                img_url = to_medium_url
                        when 'original'
                                img_url = to_original_url
                        else
                                img_url = to_thumbnail_url  
                        end
                        page_url  = to_page_url
                        return "<div class=\"photo-region\">\n<a href=\"\#{page_url}\" target=\"__page-\#{id}\">\n<img class=\"photo\" id=\"photo_\#{id}\" src=\"\#{img_url}\" title=\"\#{title}\" />\n</a><br />\n<input type=\"checkbox\" name=\"photo\" value=\"\#{id}\" />\n<label for=\"photo_\#{id}\">Select</label>\n</div>\n"
                end
